// admin.js
// Admin dashboard and exam creation logic

const Admin = {
    questionCount: 0,

    init: function() {
        this.bindEvents();
        this.renderExamsList();
        // Clear previous form state
        document.getElementById('questions-container').innerHTML = '';
        this.questionCount = 0;
        this.addQuestionBlock(); // Add one default question
    },

    bindEvents: function() {
        const addBtn = document.getElementById('add-question-btn');
        const form = document.getElementById('create-exam-form');
        
        // Remove existing listeners to prevent duplicates if init is called multiple times
        const newAddBtn = addBtn.cloneNode(true);
        addBtn.parentNode.replaceChild(newAddBtn, addBtn);
        newAddBtn.addEventListener('click', this.addQuestionBlock.bind(this));

        const newForm = form.cloneNode(true);
        form.parentNode.replaceChild(newForm, form);
        newForm.addEventListener('submit', this.handleSaveExam.bind(this));
    },

    addQuestionBlock: function() {
        this.questionCount++;
        const qId = this.questionCount;
        const container = document.getElementById('questions-container');
        
        const block = document.createElement('div');
        block.className = 'question-block';
        block.id = `qblock-${qId}`;
        
        block.innerHTML = `
            <button type="button" class="remove-question-btn" onclick="document.getElementById('qblock-${qId}').remove()">&times;</button>
            <div class="input-group">
                <label>Question ${qId}</label>
                <input type="text" class="q-text" required placeholder="Enter question text">
            </div>
            <div class="options-grid">
                <div class="input-group">
                    <label>Option A</label>
                    <input type="text" class="q-opt-a" required placeholder="Option A">
                </div>
                <div class="input-group">
                    <label>Option B</label>
                    <input type="text" class="q-opt-b" required placeholder="Option B">
                </div>
                <div class="input-group">
                    <label>Option C</label>
                    <input type="text" class="q-opt-c" required placeholder="Option C">
                </div>
                <div class="input-group">
                    <label>Option D</label>
                    <input type="text" class="q-opt-d" required placeholder="Option D">
                </div>
            </div>
            <div class="input-group">
                <label>Correct Answer</label>
                <select class="q-correct" required>
                    <option value="A">Option A</option>
                    <option value="B">Option B</option>
                    <option value="C">Option C</option>
                    <option value="D">Option D</option>
                </select>
            </div>
        `;
        
        container.appendChild(block);
    },

    handleSaveExam: function(e) {
        e.preventDefault();
        
        const title = document.getElementById('exam-title').value.trim();
        const questionBlocks = document.querySelectorAll('.question-block');
        
        if (questionBlocks.length === 0) {
            alert('Please add at least one question.');
            return;
        }

        const questions = Array.from(questionBlocks).map(block => {
            return {
                text: block.querySelector('.q-text').value.trim(),
                options: {
                    A: block.querySelector('.q-opt-a').value.trim(),
                    B: block.querySelector('.q-opt-b').value.trim(),
                    C: block.querySelector('.q-opt-c').value.trim(),
                    D: block.querySelector('.q-opt-d').value.trim()
                },
                correct: block.querySelector('.q-correct').value
            };
        });

        const exam = {
            title: title,
            questions: questions,
            createdAt: new Date().toISOString()
        };

        DB.saveExam(exam);
        
        // Reset form
        document.getElementById('create-exam-form').reset();
        document.getElementById('questions-container').innerHTML = '';
        this.questionCount = 0;
        this.addQuestionBlock();
        
        alert('Exam saved successfully!');
        this.renderExamsList();
    },

    renderExamsList: function() {
        const exams = DB.getExams();
        const list = document.getElementById('admin-exam-list');
        list.innerHTML = '';

        if (exams.length === 0) {
            list.innerHTML = '<li>No exams created yet.</li>';
            return;
        }

        exams.forEach(exam => {
            const li = document.createElement('li');
            li.innerHTML = `
                <div>
                    <strong>${exam.title}</strong>
                    <div style="font-size: 0.85em; color: var(--text-muted);">${exam.questions.length} questions</div>
                </div>
                <div>
                    <span style="font-size: 0.85em; color: var(--text-muted);">${new Date(exam.createdAt).toLocaleDateString()}</span>
                </div>
            `;
            list.appendChild(li);
        });
    }
};


// db.js
// Mock Database using LocalStorage

const DB_KEYS = {
    USERS: 'proctora_users',
    EXAMS: 'proctora_exams',
    RESULTS: 'proctora_results'
};

const DB = {
    init: function() {
        if (!localStorage.getItem(DB_KEYS.USERS)) {
            const initialUsers = [
                { id: 'admin', role: 'admin', password: 'password' },
                { id: 'student1', role: 'student', password: 'password', name: 'Student One' },
                { id: 'student2', role: 'student', password: 'password', name: 'Student Two' },
                { id: 'student3', role: 'student', password: 'password', name: 'Student Three' },
                { id: 'student4', role: 'student', password: 'password', name: 'Student Four' },
                { id: 'student5', role: 'student', password: 'password', name: 'Student Five' }
            ];
            localStorage.setItem(DB_KEYS.USERS, JSON.stringify(initialUsers));
        }

        if (!localStorage.getItem(DB_KEYS.EXAMS)) {
            localStorage.setItem(DB_KEYS.EXAMS, JSON.stringify([]));
        }

        if (!localStorage.getItem(DB_KEYS.RESULTS)) {
            localStorage.setItem(DB_KEYS.RESULTS, JSON.stringify([]));
        }
    },

    getUsers: function() {
        return JSON.parse(localStorage.getItem(DB_KEYS.USERS));
    },

    authenticate: function(username, password) {
        const users = this.getUsers();
        return users.find(u => u.id === username && u.password === password) || null;
    },

    getExams: function() {
        return JSON.parse(localStorage.getItem(DB_KEYS.EXAMS));
    },

    saveExam: function(exam) {
        const exams = this.getExams();
        exam.id = 'exam_' + Date.now();
        exams.push(exam);
        localStorage.setItem(DB_KEYS.EXAMS, JSON.stringify(exams));
    },

    getResults: function(studentId = null) {
        const results = JSON.parse(localStorage.getItem(DB_KEYS.RESULTS));
        if (studentId) {
            return results.filter(r => r.studentId === studentId);
        }
        return results;
    },

    saveResult: function(result) {
        const results = this.getResults();
        results.push(result);
        localStorage.setItem(DB_KEYS.RESULTS, JSON.stringify(results));
    }
};

// Initialize DB on script load
DB.init();


// main.js
// Routing and authentication logic

let currentUser = null;

const App = {
    views: ['login-view', 'admin-view', 'student-view', 'exam-taking-view'],
    
    init: function() {
        this.bindEvents();
        this.checkSession();
    },

    bindEvents: function() {
        document.getElementById('login-form').addEventListener('submit', this.handleLogin.bind(this));
        document.getElementById('admin-logout').addEventListener('click', this.handleLogout.bind(this));
        document.getElementById('student-logout').addEventListener('click', this.handleLogout.bind(this));
    },

    switchView: function(viewId) {
        this.views.forEach(v => {
            document.getElementById(v).classList.add('hidden');
            document.getElementById(v).classList.remove('active');
        });
        document.getElementById(viewId).classList.remove('hidden');
        document.getElementById(viewId).classList.add('active');
    },

    handleLogin: function(e) {
        e.preventDefault();
        const usernameInput = document.getElementById('username').value.trim();
        const passwordInput = document.getElementById('password').value;
        const errorMsg = document.getElementById('login-error');

        const user = DB.authenticate(usernameInput, passwordInput);

        if (user) {
            errorMsg.classList.add('hidden');
            this.loginUser(user);
        } else {
            errorMsg.classList.remove('hidden');
        }
    },

    loginUser: function(user) {
        currentUser = user;
        sessionStorage.setItem('currentUser', JSON.stringify(user));
        
        if (user.role === 'admin') {
            this.switchView('admin-view');
            Admin.init();
        } else if (user.role === 'student') {
            this.switchView('student-view');
            Student.init();
        }
    },

    handleLogout: function() {
        currentUser = null;
        sessionStorage.removeItem('currentUser');
        document.getElementById('login-form').reset();
        this.switchView('login-view');
    },

    checkSession: function() {
        const storedUser = sessionStorage.getItem('currentUser');
        if (storedUser) {
            this.loginUser(JSON.parse(storedUser));
        } else {
            this.switchView('login-view');
        }
    }
};

document.addEventListener('DOMContentLoaded', () => {
    App.init();
});
